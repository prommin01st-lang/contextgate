```typescript
import { Server } from '@modelcontextprotocol/sdk/server/index.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';
import {
  CallToolRequestSchema,
  ListResourcesRequestSchema,
  ListToolsRequestSchema,
  ReadResourceRequestSchema,
} from '@modelcontextprotocol/sdk/types.js';

export class ContextGateServer {
  private server: Server;
  private router: RequestRouter;
  private auth: ApiKeyAuth;

  constructor(options: ServerOptions) {
    this.server = new Server(
      { name: 'contextgate', version: '0.1.0' },
      { capabilities: { resources: {}, tools: {} } }
    );
    this.router = options.router;
    this.auth = options.auth;
    this.setupHandlers();
  }

  private setupHandlers() {
    // Auth middleware
    this.server.setRequestHandler(ListResourcesRequestSchema, async (req) => {
      const agent = await this.auth.validate(req.meta?.apiKey);
      return this.router.listResources(agent, req.params?._meta?.workspaceId);
    });

    this.server.setRequestHandler(ReadResourceRequestSchema, async (req) => {
      const agent = await this.auth.validate(req.meta?.apiKey);
      await this.checkPolicy(agent, req.params.uri, 'read');
      return this.router.readResource(agent, req.params.uri);
    });

    this.server.setRequestHandler(ListToolsRequestSchema, async (req) => {
      const agent = await this.auth.validate(req.meta?.apiKey);
      return this.router.listTools(agent);
    });

    this.server.setRequestHandler(CallToolRequestSchema, async (req) => {
      const agent = await this.auth.validate(req.meta?.apiKey);
      await this.checkPolicy(agent, req.params.name, 'use');
      return this.router.callTool(agent, req.params.name, req.params.arguments);
    });
  }

  async startStdio() {
    const transport = new StdioServerTransport();
    await this.server.connect(transport);
  }

  async startSSE(port: number) {
    // Implement SSE transport with Hono
  }
}
```

### 1.3 Connectors (Week 2-3)

#### Step 7: FileSystem Connector

**apps/server/src/connectors/filesystem.ts**:
```typescript
import { BaseConnector, MCPResource, MCPTool } from '@contextgate/connectors';
import { promises as fs } from 'fs';
import * as path from 'path';
import { glob } from 'glob';

interface FileSystemConfig {
  basePath: string;
  includePatterns: string[]; // ['**/*.md', '**/*.txt']
  excludePatterns?: string[];
}

export class FileSystemConnector extends BaseConnector {
  private resolvedPath: string;

  async connect(): Promise<void> {
    const config = this.config.config as FileSystemConfig;
    this.resolvedPath = path.resolve(config.basePath);
    const stat = await fs.stat(this.resolvedPath);
    if (!stat.isDirectory()) {
      throw new Error(`Path ${this.resolvedPath} is not a directory`);
    }
  }

  async listResources(): Promise<MCPResource[]> {
    const config = this.config.config as FileSystemConfig;
    const files = await glob(config.includePatterns, {
      cwd: this.resolvedPath,
      ignore: config.excludePatterns || [],
    });

    return files.map(file => ({
      uri: `filesystem://${this.config.id}/${file}`,
      name: file,
      mimeType: this.getMimeType(file),
    }));
  }

  async readResource(uri: string): Promise<MCPResource> {
    const filePath = uri.replace(`filesystem://${this.config.id}/`, '');
    const fullPath = path.join(this.resolvedPath, filePath);
    
    // Security: ensure path is within basePath
    if (!fullPath.startsWith(this.resolvedPath)) {
      throw new Error('Access denied: path traversal detected');
    }

    const content = await fs.readFile(fullPath, 'utf-8');
    return {
      uri,
      name: filePath,
      mimeType: this.getMimeType(filePath),
      text: content,
    };
  }

  async listTools(): Promise<MCPTool[]> {
    return [{
      name: `search_files_${this.config.id}`,
      description: `Search files in ${this.config.name}`,
      inputSchema: {
        type: 'object',
        properties: {
          query: { type: 'string', description: 'Search term' },
        },
        required: ['query'],
      },
    }];
  }

  async callTool(name: string, args: unknown): Promise<unknown> {
    // Implement file search
    return { message: 'Not implemented in MVP' };
  }

  private getMimeType(file: string): string {
    const ext = path.extname(file);
    const mimeTypes: Record<string, string> = {
      '.md': 'text/markdown',
      '.txt': 'text/plain',
      '.json': 'application/json',
      '.csv': 'text/csv',
    };
    return mimeTypes[ext] || 'application/octet-stream';
  }
}
```

#### Step 8: PostgreSQL Connector (Read-Only)

**Dependencies**: `pg`, `@types/pg`

**apps/server/src/connectors/postgres.ts**:
```typescript
import { BaseConnector, MCPResource, MCPTool } from '@contextgate/connectors';
import { Pool, PoolClient } from 'pg';

interface PostgresConfig {
  host: string;
  port: number;
  database: string;
  user: string;
  password: string;
  ssl?: boolean;
  allowedTables?: string[];
  readOnly?: boolean;
}

export class PostgresConnector extends BaseConnector {
  private pool: Pool | null = null;

  async connect(): Promise<void> {
    const config = this.config.config as PostgresConfig;
    this.pool = new Pool({
      host: config.host,
      port: config.port,
      database: config.database,
      user: config.user,
      password: config.password,
      ssl: config.ssl ? { rejectUnauthorized: false } : false,
    });

    // Verify connection
    const client = await this.pool.connect();
    await client.query('SELECT 1');
    client.release();
  }

  async disconnect(): Promise<void> {
    await this.pool?.end();
  }

  async listResources(): Promise<MCPResource[]> {
    const config = this.config.config as PostgresConfig;
    const client = await this.pool!.connect();
    try {
      const result = await client.query(`
        SELECT table_name 
        FROM information_schema.tables 
        WHERE table_schema = 'public'
        ${config.allowedTables ? `AND table_name = ANY($1)` : ''}
      `, config.allowedTables ? [config.allowedTables] : []);

      return result.rows.map(row => ({
        uri: `postgres://${this.config.id}/table/${row.table_name}`,
        name: row.table_name,
        mimeType: 'application/json',
      }));
    } finally {
      client.release();
    }
  }

  async readResource(uri: string): Promise<MCPResource> {
    const match = uri.match(/postgres:\/\/[^/]+\/table\/(.+)/);
    if (!match) throw new Error('Invalid resource URI');
    
    const tableName = match[1];
    const config = this.config.config as PostgresConfig;
    
    if (config.allowedTables && !config.allowedTables.includes(tableName)) {
      throw new Error('Table not allowed');
    }

    const client = await this.pool!.connect();
    try {
      // Get schema
      const schemaResult = await client.query(`
        SELECT column_name, data_type 
        FROM information_schema.columns 
        WHERE table_name = $1
      `, [tableName]);

      // Get sample data (limit 10)
      const dataResult = await client.query(
        `SELECT * FROM "${tableName}" LIMIT 10`
      );

      return {
        uri,
        name: tableName,
        mimeType: 'application/json',
        text: JSON.stringify({
          schema: schemaResult.rows,
          sample: dataResult.rows,
        }, null, 2),
      };
    } finally {
      client.release();
    }
  }

  async listTools(): Promise<MCPTool[]> {
    return [{
      name: `query_${this.config.id}`,
      description: `Run read-only SQL query on ${this.config.name}`,
      inputSchema: {
        type: 'object',
        properties: {
          sql: { 
            type: 'string', 
            description: 'SQL SELECT query (read-only)' 
          },
        },
        required: ['sql'],
      },
    }];
  }

  async callTool(name: string, args: unknown): Promise<unknown> {
    const config = this.config.config as PostgresConfig;
    if (config.readOnly !== false) {
      const sql = (args as { sql: string }).sql.toLowerCase();
      if (sql.match(/\b(insert|update|delete|drop|alter|create|truncate)\b/)) {
        throw new Error('Write operations not allowed in read-only mode');
      }
    }

    const client = await this.pool!.connect();
    try {
      const result = await client.query((args as { sql: string }).sql);
      return {
        content: [{ type: 'text', text: JSON.stringify(result.rows, null, 2) }],
      };
    } finally {
      client.release();
    }
  }
}
```

```typescript
      event: 'endpoint',

      clearInterval(interval);
      mcpServer.close();
    });
  });
```

### 1.5 Policy Engine (Week 3)

#### Step 10: RBAC Implementation

**apps/server/src/policies/engine.ts**:
```typescript
import { db } from '../db';
import { policies, agents } from '../db/schema';
import { eq, and, like } from 'drizzle-orm';

interface PolicyCheck {
  agentId: string;
  resourceUri: string;
  action: 'read' | 'list' | 'query' | 'use';
}

export class PolicyEngine {
  async check(request: PolicyCheck): Promise<boolean> {
    const agent = await db.query.agents.findFirst({
      where: eq(agents.id, request.agentId),
    });

    if (!agent || !agent.isActive) return false;

    // Get applicable policies (ordered by specificity)
    const agentPolicies = await db.select()
      .from(policies)
      .where(
        and(
          eq(policies.agentId, request.agentId),
          eq(policies.action, request.action)
        )
      );

    // Check deny policies first
    const denyPolicy = agentPolicies.find(p => 
      !p.isAllowed && this.matchPattern(request.resourceUri, p.resourcePattern)
    );
    if (denyPolicy) return false;

    // Check allow policies
    const allowPolicy = agentPolicies.find(p => 
      p.isAllowed && this.matchPattern(request.resourceUri, p.resourcePattern)
    );
    return !!allowPolicy;
  }

  private matchPattern(uri: string, pattern: string): boolean {
    // Support glob patterns: filesystem://*/*.md
    const regex = new RegExp('^' + pattern.replace(/\*/g, '.*') + '$');
    return regex.test(uri);
  }
}
```

### 2.3 Notion Connector (Week 2)

**Dependencies**: `@notionhq/client`, `grammy`, `google-spreadsheet`, `google-auth-library`

**apps/server/src/connectors/notion.ts**:
```typescript
import { Client as NotionClient } from '@notionhq/client';
import { BaseConnector } from '@contextgate/connectors';

interface NotionConfig {
  token: string;
  databaseIds?: string[];
  pageIds?: string[];
}

export class NotionConnector extends BaseConnector {
  private client: NotionClient;

  async connect(): Promise<void> {
    const config = this.config.config as NotionConfig;
    this.client = new NotionClient({ auth: config.token });
  }

  async listResources(): Promise<MCPResource[]> {
    const resources: MCPResource[] = [];
    
    // List databases
    const databases = await this.client.search({
      filter: { property: 'object', value: 'database' },
    });

    for (const db of databases.results) {
      if ('title' in db) {
        resources.push({
          uri: `notion://${this.config.id}/database/${db.id}`,
          name: db.title[0]?.plain_text || 'Untitled Database',
          mimeType: 'application/json',
        });
      }
    }

    // List pages
    const pages = await this.client.search({
      filter: { property: 'object', value: 'page' },
    });

    for (const page of pages.results) {
      const pageDetails = await this.client.pages.retrieve({ page_id: page.id });
      resources.push({
        uri: `notion://${this.config.id}/page/${page.id}`,
        name: this.extractPageTitle(pageDetails),
        mimeType: 'text/markdown',
      });
    }

    return resources;
  }

  async readResource(uri: string): Promise<MCPResource> {
    const match = uri.match(/notion:\/\/[^/]+\/(database|page)\/(.+)/);
    if (!match) throw new Error('Invalid Notion URI');

    const [, type, id] = match;

    if (type === 'page') {
      const blocks = await this.client.blocks.children.list({ block_id: id });
      const markdown = this.blocksToMarkdown(blocks.results);
      return { uri, name: 'Page', mimeType: 'text/markdown', text: markdown };
    }

    if (type === 'database') {
      const rows = await this.client.databases.query({ database_id: id });
      return { 
        uri, 
        name: 'Database', 
        mimeType: 'application/json', 
        text: JSON.stringify(rows.results, null, 2) 
      };
    }

    throw new Error('Unknown Notion resource type');
  }

  private blocksToMarkdown(blocks: any[]): string {
    return blocks.map(block => {
      switch (block.type) {
        case 'paragraph':
          return block.paragraph.rich_text.map((t: any) => t.plain_text).join('');
        case 'heading_1':
          return `# ${block.heading_1.rich_text.map((t: any) => t.plain_text).join('')}`;
        case 'heading_2':
          return `## ${block.heading_2.rich_text.map((t: any) => t.plain_text).join('')}`;
        case 'bulleted_list_item':
          return `- ${block.bulleted_list_item.rich_text.map((t: any) => t.plain_text).join('')}`;
        default:
          return '';
      }
    }).join('\n\n');
  }

  private extractPageTitle(page: any): string {
    if ('properties' in page && page.properties?.title?.title) {
      return page.properties.title.title[0]?.plain_text || 'Untitled';
    }
    return 'Untitled';
  }
}
```

```typescript
import { BaseConnector, MCPResource, MCPTool } from '@contextgate/connectors';
import { Bot } from 'grammy';

interface TelegramConfig {
  botToken: string;
  allowedChats?: number[]; // Chat IDs whitelist
  messageLimit?: number;   // Messages per read, default 50
}

export class TelegramConnector extends BaseConnector {
  private bot: Bot;
  private tgConfig: TelegramConfig;

  async connect(): Promise<void> {
    this.tgConfig = this.config.config as TelegramConfig;
    this.bot = new Bot(this.tgConfig.botToken);
    await this.bot.api.getMe(); // Validate token
  }

  async disconnect(): Promise<void> {
    // Grammy cleanup
  }

  async listResources(): Promise<MCPResource[]> {
    const updates = await this.bot.api.getUpdates({ limit: 100 });
    const chatSet = new Map<number, string>();

    for (const update of updates) {
      const chat = update.message?.chat || update.channel_post?.chat;
      if (chat && (!this.tgConfig.allowedChats || this.tgConfig.allowedChats.includes(chat.id))) {
        chatSet.set(chat.id, chat.title || chat.username || `Chat ${chat.id}`);
      }
    }

    return Array.from(chatSet.entries()).map(([id, name]) => ({
      uri: `telegram://${this.config.id}/chat/${id}`,
      name,
      mimeType: 'text/plain',
    }));
  }

  async readResource(uri: string): Promise<MCPResource> {
    const match = uri.match(/telegram:\/\/[^/]+\/chat\/(\d+)/);
    if (!match) throw new Error('Invalid Telegram URI');
    const chatId = Number(match[1]);

    if (this.tgConfig.allowedChats && !this.tgConfig.allowedChats.includes(chatId)) {
      throw new Error('Chat not allowed by policy');
    }

    const limit = this.tgConfig.messageLimit || 50;
    const messages = await this.bot.api.getUpdates({ limit: 100 });

    const chatMessages = messages
      .filter(u => (u.message?.chat.id === chatId) || (u.channel_post?.chat.id === chatId))
      .slice(-limit)
      .map(u => {
        const msg = u.message || u.channel_post;
        const date = new Date(msg.date * 1000).toISOString();
        const sender = msg.from?.username || msg.from?.first_name || 'Unknown';
        return `[${date}] ${sender}: ${msg.text || '(media)'}`;
      });

    return {
      uri,
      name: `Telegram Chat ${chatId}`,
      mimeType: 'text/plain',
      text: chatMessages.join('\n'),
    };
  }

  async listTools(): Promise<MCPTool[]> {
    return [{
      name: `send_message_${this.config.id}`,
      description: `Send a message to a Telegram chat via ${this.config.name}`,
      inputSchema: {
        type: 'object',
        properties: {
          chatId: { type: 'number', description: 'Target chat ID' },
          message: { type: 'string', description: 'Message text to send' },
        },
        required: ['chatId', 'message'],
      },
    }];
  }

  async callTool(name: string, args: unknown): Promise<unknown> {
    const { chatId, message } = args as { chatId: number; message: string };

    if (this.tgConfig.allowedChats && !this.tgConfig.allowedChats.includes(chatId)) {
      throw new Error('Sending to this chat is not allowed');
    }

    const result = await this.bot.api.sendMessage(chatId, message);
    return {
      content: [{ type: 'text', text: `Sent message. Message ID: ${result.message_id}` }],
    };
  }
}
```

```typescript
import { BaseConnector, MCPResource, MCPTool } from '@contextgate/connectors';
import { GoogleSpreadsheet } from 'google-spreadsheet';
import { JWT } from 'google-auth-library';

interface GoogleSheetsConfig {
  serviceAccountEmail: string;
  privateKey: string;
  spreadsheetIds: string[]; // Whitelist
  readOnly?: boolean;       // default: true
}

export class GoogleSheetsConnector extends BaseConnector {
  private gsConfig: GoogleSheetsConfig;

  async connect(): Promise<void> {
    this.gsConfig = this.config.config as GoogleSheetsConfig;
    // Validate by loading first spreadsheet
    const auth = new JWT({
      email: this.gsConfig.serviceAccountEmail,
      key: this.gsConfig.privateKey,
      scopes: ['https://www.googleapis.com/auth/spreadsheets.readonly'],
    });
    const doc = new GoogleSpreadsheet(this.gsConfig.spreadsheetIds[0], auth);
    await doc.loadInfo();
  }

  async listResources(): Promise<MCPResource[]> {
    const auth = new JWT({
      email: this.gsConfig.serviceAccountEmail,
      key: this.gsConfig.privateKey,
      scopes: ['https://www.googleapis.com/auth/spreadsheets.readonly'],
    });

    const resources: MCPResource[] = [];
    for (const id of this.gsConfig.spreadsheetIds) {
      const doc = new GoogleSpreadsheet(id, auth);
      await doc.loadInfo();
      resources.push({
        uri: `gsheets://${this.config.id}/spreadsheet/${id}`,
        name: doc.title || `Spreadsheet ${id}`,
        mimeType: 'application/json',
      });

      for (const sheet of doc.sheetsByIndex) {
        resources.push({
          uri: `gsheets://${this.config.id}/sheet/${id}/${sheet.sheetId}`,
          name: `${doc.title} / ${sheet.title}`,
          mimeType: 'text/csv',
        });
      }
    }
    return resources;
  }

  async readResource(uri: string): Promise<MCPResource> {
    const match = uri.match(/gsheets:\/\/[^/]+\/(spreadsheet|sheet)\/([^/]+)(?:\/(\d+))?/);
    if (!match) throw new Error('Invalid Google Sheets URI');
    const [, type, spreadsheetId, sheetId] = match;

    if (!this.gsConfig.spreadsheetIds.includes(spreadsheetId)) {
      throw new Error('Spreadsheet not in allowed list');
    }

    const auth = new JWT({
      email: this.gsConfig.serviceAccountEmail,
      key: this.gsConfig.privateKey,
      scopes: ['https://www.googleapis.com/auth/spreadsheets.readonly'],
    });
    const doc = new GoogleSpreadsheet(spreadsheetId, auth);
    await doc.loadInfo();

    if (type === 'spreadsheet') {
      const metadata = {
        title: doc.title,
        sheets: doc.sheetsByIndex.map(s => ({ id: s.sheetId, title: s.title, rowCount: s.rowCount })),
      };
      return {
        uri,
        name: doc.title,
        mimeType: 'application/json',
        text: JSON.stringify(metadata, null, 2),
      };
    }

    if (type === 'sheet' && sheetId) {
      const sheet = doc.sheetsById[Number(sheetId)];
      if (!sheet) throw new Error('Sheet not found');

      await sheet.loadCells();
      const rows: string[][] = [];
      for (let r = 0; r < sheet.rowCount; r++) {
        const row: string[] = [];
        for (let c = 0; c < sheet.columnCount; c++) {
          const cell = sheet.getCell(r, c);
          row.push(cell.value?.toString() || '');
        }
        rows.push(row);
      }

      const csv = rows.map(r => r.join(',')).join('\n');
      return {
        uri,
        name: `${doc.title} / ${sheet.title}`,
        mimeType: 'text/csv',
        text: csv,
      };
    }

    throw new Error('Unknown resource type');
  }

  async listTools(): Promise<MCPTool[]> {
    const tools: MCPTool[] = [];
    if (!this.gsConfig.readOnly) {
      tools.push({
        name: `append_row_${this.config.id}`,
        description: 'Append a row to a Google Sheet',
        inputSchema: {
          type: 'object',
          properties: {
            spreadsheetId: { type: 'string' },
            sheetId: { type: 'number' },
            row: { type: 'array', items: { type: 'string' }, description: 'Row values' },
          },
          required: ['spreadsheetId', 'sheetId', 'row'],
        },
      });
    }
    return tools;
  }

  async callTool(name: string, args: unknown): Promise<unknown> {
    if (this.gsConfig.readOnly !== false) {
      throw new Error('Write operations not allowed: connector is read-only');
    }

    const { spreadsheetId, sheetId, row } = args as { spreadsheetId: string; sheetId: number; row: string[] };

    const auth = new JWT({
      email: this.gsConfig.serviceAccountEmail,
      key: this.gsConfig.privateKey,
      scopes: ['https://www.googleapis.com/auth/spreadsheets'],
    });
    const doc = new GoogleSpreadsheet(spreadsheetId, auth);
    await doc.loadInfo();
    const sheet = doc.sheetsById[sheetId];
    if (!sheet) throw new Error('Sheet not found');

    await sheet.addRow(row);
    return {
      content: [{ type: 'text', text: 'Row appended successfully' }],
    };
  }
}
```

```typescript
import { BaseConnector, MCPTool } from '@contextgate/connectors';
import SwaggerParser from '@apidevtools/swagger-parser';
import { OpenAPIV3 } from 'openapi-types';
import { ky } from 'ky';

interface RestApiConfig {
  baseUrl: string;
  openApiSpec?: string; // URL or JSON string
  headers?: Record<string, string>;
  auth?: {
    type: 'bearer' | 'apiKey';
    value: string;
  };
}

export class RestApiConnector extends BaseConnector {
  private spec: OpenAPIV3.Document | null = null;

  async connect(): Promise<void> {
    const config = this.config.config as RestApiConfig;
    if (config.openApiSpec) {
      this.spec = await SwaggerParser.validate(config.openApiSpec);
    }
  }

  async listTools(): Promise<MCPTool[]> {
    if (!this.spec) return [];

    const tools: MCPTool[] = [];
    for (const [path, methods] of Object.entries(this.spec.paths || {})) {
      for (const [method, operation] of Object.entries(methods || {})) {
        if (typeof operation === 'object' && 'operationId' in operation) {
          tools.push({
            name: `api_${this.config.id}_${operation.operationId}`,
            description: operation.summary || `${method} ${path}`,
            inputSchema: this.convertSchema(operation.requestBody),
          });
        }
      }
    }
    return tools;
  }

  async callTool(name: string, args: unknown): Promise<unknown> {
    const config = this.config.config as RestApiConfig;
    const operationId = name.replace(`api_${this.config.id}_`, '');
    
    // Find operation by operationId
    let targetPath = '';
    let targetMethod = '';
    
    for (const [path, methods] of Object.entries(this.spec?.paths || {})) {
      for (const [method, operation] of Object.entries(methods || {})) {
        if (typeof operation === 'object' && operation.operationId === operationId) {
          targetPath = path;
          targetMethod = method;
          break;
        }
      }
    }

    // Build request
    const url = new URL(targetPath.replace(/{(\w+)}/g, (_, key) => (args as any)[key]), config.baseUrl);
    const response = await fetch(url, {
      method: targetMethod.toUpperCase(),
      headers: {
        'Content-Type': 'application/json',
        ...(config.auth?.type === 'bearer' ? { Authorization: `Bearer ${config.auth.value}` } : {}),
        ...(config.headers || {}),
      },
      body: ['get', 'head'].includes(targetMethod) ? undefined : JSON.stringify(args),
    });

    return response.json();
  }

  private convertSchema(requestBody: any): object {
    if (!requestBody?.content?.['application/json']?.schema) {
      return { type: 'object', properties: {} };
    }
    return requestBody.content['application/json'].schema;
  }
}
```

```typescript
export class AdvancedPolicyEngine extends PolicyEngine {
  async check(request: PolicyCheck & { timestamp?: Date }): Promise<boolean> {
    const time = request.timestamp || new Date();
    
    const policies = await this.getApplicablePolicies(request.agentId);
    
    for (const policy of policies) {
      // Time-based check
      if (policy.timeStart && policy.timeEnd) {
        const currentTime = `${time.getHours()}:${time.getMinutes()}`;
        if (currentTime < policy.timeStart || currentTime > policy.timeEnd) {
          continue; // Skip this policy
        }
      }

      // Day of week check
      if (policy.daysOfWeek && !policy.daysOfWeek.includes(time.getDay())) {
        continue;
      }

      if (this.matchPattern(request.resourceUri, policy.resourcePattern)) {
        return policy.isAllowed;
      }
    }

    return false; // Default deny
  }
}
```

```typescript
import { pathToFileURL } from 'url';
import { BaseConnector } from '@contextgate/connectors';

export class ConnectorLoader {
  async loadFromNpm(packageName: string): Promise<typeof BaseConnector> {
    // Install if not exists
    const { execSync } = require('child_process');
    execSync(`npm install ${packageName}`, { cwd: process.cwd() });
    
    // Dynamic import
    const module = await import(packageName);
    return module.default || module;
  }

  async loadFromPath(filePath: string): Promise<typeof BaseConnector> {
    const module = await import(pathToFileURL(filePath).href);
    return module.default || module;
  }
}
```

### 3.4 Webhook / Event-Driven Sync (Week 4-5)

**apps/server/src/webhooks/index.ts**:
```typescript
import { Hono } from 'hono';
import { db } from '../db';
import { syncQueue } from '../jobs';

const webhooks = new Hono();

// Notion webhook
webhooks.post('/notion/:connectorId', async (c) => {
  const connectorId = c.req.param('connectorId');
  const signature = c.req.header('x-notion-signature');
  
  // Verify signature
  const isValid = await verifyNotionSignature(signature, await c.req.raw.clone().text());
  if (!isValid) return c.json({ error: 'Invalid signature' }, 401);

  // Trigger sync
  await syncQueue.add('webhook-sync', { connectorId });
  
  return c.json({ received: true });
});

// Generic webhook
webhooks.post('/:workspaceId/:connectorId', async (c) => {
  const { workspaceId, connectorId } = c.req.param();
  const body = await c.req.json();
  
  // Store webhook event
  await db.insert(webhookEvents).values({
    workspaceId,
    connectorId,
    payload: body,
    createdAt: new Date(),
  });

  // Trigger immediate sync
  await syncQueue.add('webhook-sync', { connectorId, payload: body });

  return c.json({ received: true });
});
```

### 3.5 Slack & GitHub Connectors (Week 5)

**Slack Connector**:
```typescript
import { WebClient } from '@slack/web-api';

export class SlackConnector extends BaseConnector {
  private client: WebClient;

  async connect(): Promise<void> {
    const config = this.config.config as { token: string; channels?: string[] };
    this.client = new WebClient(config.token);
  }

  async listResources(): Promise<MCPResource[]> {
    const channels = await this.client.conversations.list();
    return (channels.channels || [])
      .filter(ch => ch.is_channel)
      .map(ch => ({
        uri: `slack://${this.config.id}/channel/${ch.id}`,
        name: ch.name || 'unknown',
        mimeType: 'application/json',
      }));
  }

  async readResource(uri: string): Promise<MCPResource> {
    const match = uri.match(/slack:\/\/[^/]+\/channel\/(.+)/);
    if (!match) throw new Error('Invalid Slack URI');
    
    const channelId = match[1];
    const history = await this.client.conversations.history({ channel: channelId, limit: 50 });
    
    const formatted = history.messages?.map(m => 
      `[${new Date(Number(m.ts) * 1000).toISOString()}] ${m.user}: ${m.text}`
    ).join('\n');

    return {
      uri,
      name: `Channel History`,
      mimeType: 'text/plain',
      text: formatted || 'No messages',
    };
  }
}
```
